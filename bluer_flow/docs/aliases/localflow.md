# localflow

```bash
localflow \
	eval \
	[type=<cpu|gpu>,verbose] \
	<command-line>
 . <command-line> -> localflow
localflow \
	list \
	[status=<status>]
 . list localflow jobs.
localflow \
	rm \
	[status=<status>]
 . rm localflow jobs.
localflow \
	start \
	[exit_if_no_job]
 . start localflow
localflow \
	stop
 . stop localflow
```
